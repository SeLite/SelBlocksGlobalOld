function assertFunctionFailure func
    getEval $hasFailed= false
    try
        call ${func}
    catch
        getEval $hasFailed= true
    endTry
    assertEval $hasFailed true
endFunction
function assertFunctionFailure_test
    getEval throw new Error('test') regexp:1
endFunction
call assertFunctionFailure func='assertFunctionFailure_test'
